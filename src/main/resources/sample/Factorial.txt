MODULE Factorial;
FROM Terminal2 IMPORT WriteCard, WriteLn;

VAR factorial, Index, n : CARDINAL;

PROCEDURE Fact(n:CARDINAL);
BEGIN
	factorial := 1;
	FOR n:=n TO 1 BY -1 DO
	    factorial:=factorial*n;
	END;
END Fact;

BEGIN
   FOR Index := 0 TO 8 DO
	WriteCard(Index,3);
	Fact(Index);
	WriteCard(factorial,12);
	WriteLn;
   END;
END Factorial.
